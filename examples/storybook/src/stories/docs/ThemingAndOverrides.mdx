  <DocsSection
    title="Which override layer to use"
    description="Prefer the narrowest layer that solves the problem. Broader layers have more visual blast radius."
  >
    <DocsGrid>
      <DocsCard title="Preset default">
        Use the default preset when the existing GoodWidget visual system already matches your needs.
      </DocsCard>
      <DocsCard title="Token override">
        Use `config.tokens` or `themeOverrides.tokens` for broad palette and scale changes that
        should cascade through derived themes.
      </DocsCard>
      <DocsCard title="Theme override">
        Use named theme keys for targeted surface changes such as `dark_ClaimCard` or
        `light_ClaimActionGlow`.
      </DocsCard>
      <DocsCard title="Inline props">
        Use inline styling only for one-off local treatment. It is not the host-facing design
        system contract.
      </DocsCard>
    </DocsGrid>
  </DocsSection>

  <DocsSection
    title="Host-side example"
    description="This is the integration shape to copy when a host app wants to keep widget logic unchanged but align visuals to its brand."
  >
    <Source
      dark
      language="tsx"
      code={`import { ClaimWidget } from '@goodwidget/claim-widget-theme-demo'

const themeOverrides = {
  tokens: {
    color: {
      primary: '#2E5DE8',
      primaryDark: '#1D3EB2',
      primaryLight: '#6E8DFF',
    },
  },
  themes: {
    dark_ClaimCard: {
      borderColor: '#2E5DE8',
    },
    dark_ClaimActionGlow: {
      primary: '#4F7DFF',
      primaryLight: '#9DB4FF',
    },
  },
}

export function RewardsPanel() {
  return <ClaimWidget defaultTheme="dark" themeOverrides={themeOverrides} />
}`}
    />
